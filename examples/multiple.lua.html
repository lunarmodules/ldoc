-- function with inline return and errors
-- @string name
function mul4 (name)
    if type(name) ~= 'string' then
        --- @error[1] not a string
        return nil, 'not a string'
    end
    if #name == 0 then
        --- @error[2] zero-length string
        return nil, 'zero-length string'
    end
    --- @treturn string converted to uppercase
    return name:upper()
end
